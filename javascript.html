// Comprehensive JavaScript Example

// Variables and Data Types
let name = "John";              // String
const age = 25;                 // Number (constant)
var isActive = true;            // Boolean
let hobbies = ["reading", "gaming", "coding"];  // Array
let person = {                  // Object
    firstName: "Jane",
    lastName: "Doe",
    age: 30
};

// Template Literals
console.log(`Hello, my name is ${name} and I'm ${age} years old.`);

// Functions - Regular Function
function calculateArea(width, height) {
    return width * height;
}

// Functions - Arrow Function
const multiply = (a, b) => a * b;

// Functions - Anonymous Function
const divide = function(a, b) {
    return a / b;
};

console.log("Area:", calculateArea(5, 10));
console.log("Multiply:", multiply(4, 7));
console.log("Divide:", divide(20, 4));

// Conditional Statements
if (age >= 18) {
    console.log("Adult");
} else if (age >= 13) {
    console.log("Teenager");
} else {
    console.log("Child");
}

// Switch Statement
const day = "Monday";
switch (day) {
    case "Monday":
        console.log("Start of work week");
        break;
    case "Friday":
        console.log("TGIF!");
        break;
    default:
        console.log("Regular day");
}

// Loops - For Loop
for (let i = 0; i < 5; i++) {
    console.log(`Count: ${i}`);
}

// Loops - For...of (Arrays)
for (const hobby of hobbies) {
    console.log(`Hobby: ${hobby}`);
}

// Loops - For...in (Objects)
for (const key in person) {
    console.log(`${key}: ${person[key]}`);
}

// Loops - While Loop
let counter = 0;
while (counter < 3) {
    console.log(`While counter: ${counter}`);
    counter++;
}

// Array Methods
const numbers = [1, 2, 3, 4, 5];

// Map - Transform each element
const doubled = numbers.map(num => num * 2);
console.log("Doubled:", doubled);

// Filter - Get elements that match condition
const evenNumbers = numbers.filter(num => num % 2 === 0);
console.log("Even numbers:", evenNumbers);

// Reduce - Combine all elements into single value
const sum = numbers.reduce((total, num) => total + num, 0);
console.log("Sum:", sum);

// Find - Get first element that matches condition
const found = numbers.find(num => num > 3);
console.log("First number > 3:", found);

// Object Methods
const car = {
    brand: "Toyota",
    model: "Camry",
    year: 2023,
    
    // Method inside object
    getInfo() {
        return `${this.brand} ${this.model} (${this.year})`;
    },
    
    // Arrow function (doesn't have its own 'this')
    getAge: () => {
        return new Date().getFullYear() - car.year;
    }
};

console.log("Car info:", car.getInfo());
console.log("Car age:", car.getAge());

// Destructuring - Arrays
const [first, second, ...rest] = hobbies;
console.log("First hobby:", first);
console.log("Remaining hobbies:", rest);

// Destructuring - Objects
const { firstName, lastName } = person;
console.log(`Name: ${firstName} ${lastName}`);

// Spread Operator
const moreHobbies = ["swimming", "dancing"];
const allHobbies = [...hobbies, ...moreHobbies];
console.log("All hobbies:", allHobbies);

// Rest Parameters
function sum(...args) {
    return args.reduce((total, num) => total + num, 0);
}
console.log("Sum of 1,2,3,4,5:", sum(1, 2, 3, 4, 5));

// Promises and Async/Await
function fetchData() {
    return new Promise((resolve, reject) => {
        setTimeout(() => {
            resolve("Data fetched successfully!");
        }, 1000);
    });
}

// Using Promises
fetchData()
    .then(data => console.log("Promise result:", data))
    .catch(error => console.error("Error:", error));

// Using Async/Await
async function getData() {
    try {
        const data = await fetchData();
        console.log("Async result:", data);
    } catch (error) {
        console.error("Async error:", error);
    }
}

getData();

// Classes
class Animal {
    constructor(name, species) {
        this.name = name;
        this.species = species;
    }
    
    speak() {
        console.log(`${this.name} makes a sound`);
    }
    
    // Static method
    static getKingdom() {
        return "Animalia";
    }
}

class Dog extends Animal {
    constructor(name, breed) {
        super(name, "Canine");
        this.breed = breed;
    }
    
    speak() {
        console.log(`${this.name} barks`);
    }
    
    wagTail() {
        console.log(`${this.name} wags tail`);
    }
}

const myDog = new Dog("Buddy", "Golden Retriever");
myDog.speak();
myDog.wagTail();
console.log("Kingdom:", Animal.getKingdom());

// Error Handling
try {
    const result = riskyOperation();
    console.log("Success:", result);
} catch (error) {
    console.error("Caught error:", error.message);
} finally {
    console.log("Cleanup code runs regardless");
}

function riskyOperation() {
    if (Math.random() > 0.5) {
        throw new Error("Something went wrong!");
    }
    return "Success!";
}
